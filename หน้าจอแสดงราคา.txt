    <script>
        // ฟังก์ชันสำหรับดึงข้อมูลราคาทองคำ
        async function fetchGoldPrice() {
            try {
                // หมายเหตุ: URL นี้เป็น Public API พื้นฐานสำหรับการทดสอบ 
                // หากต้องการความเสถียร 100% ควรใช้ API ที่คุณพัฒนาขึ้นเองในเซิร์ฟเวอร์หลังบ้าน
                const response = await fetch('https://api.thaigold.info/v1/latest'); 
                const data = await response.json();
                
                if(data && data.response) {
                    const price = data.response.price;
                    
                    // ใส่ลูกน้ำ (Comma) ให้ตัวเลขดูอ่านง่ายขึ้น
                    const formatNumber = (num) => Number(num).toLocaleString('th-TH');

                    document.getElementById('bar-buy').innerText = formatNumber(price.gold_bar.buy);
                    document.getElementById('bar-sell').innerText = formatNumber(price.gold_bar.sell);
                    document.getElementById('ornament-buy').innerText = formatNumber(price.gold_jewelry.buy);
                    document.getElementById('ornament-sell').innerText = formatNumber(price.gold_jewelry.sell);
                    
                    document.getElementById('update-time').innerText = "อัปเดตข้อมูลล่าสุด: " + data.response.update_time;
                }
            } catch (error) {
                console.error('ไม่สามารถดึงข้อมูลราคาทองได้:', error);
                document.getElementById('update-time').innerText = "สถานะ: รอการเชื่อมต่อข้อมูล...";
            }
        }

        // เรียกใช้ฟังก์ชันทันทีที่เปิดจอ
        fetchGoldPrice();
        
        // ตั้งค่าให้รีเฟรชราคาอัตโนมัติทุกๆ 3 นาที (180,000 มิลลิวินาที)
        setInterval(fetchGoldPrice, 180000); 
    </script>
